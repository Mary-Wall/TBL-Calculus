\begin{itemize}
\item $f(x) = \frac{x-6}{x}$
\item $y = \frac{x-6}{x}$
\item $xy = \frac{x-6}{x} x$
\item $xy + x = x-6 +x$
\item $xy + x = -6$
\item $x(y+1) = -6$

\end{itemize}
